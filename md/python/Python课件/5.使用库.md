import xlrd
# 1. 打开 xlsx 文件
xlsx = xlrd.open_workbook('d:/test.xlsx')
# 2. 获取 0 号标签页. (当前只有一个标签页)
table = xlsx.sheet_by_index(0)

nrows = table.nrows
# 4. 遍历数据
count = 0

for i in range(1, nrows):
# 使用 cell_value(row, col) 获取到指定坐标单元格的值.

total += table.cell_value(i, 2)
count += 1
print(f'平均分: {total / count}')
